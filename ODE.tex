\question
Let $y: \mathbb{R}\rightarrow \mathbb{R}$  be differentiable and satisfy the ODE: $$\frac{dy}{dx}=f(y),\quad  x \in \mathbb{R} $$ $$y(0)=y(1)=0$$  
Where, $f:\mathbb{R}\rightarrow \mathbb{R}$ is a lipschitz condition function. Then

\begin{choices}
\choice $y(x)=0$ if and only if $x\in {0,1} $
\choice $y$ is bounded.  
\choice $y$ is strictly increasing.
\choice $\frac{dy}{dx}$ is unbounded.
\end{choices}

\question 
For $\lambda \in \mathbb{R}$, consider the boundary value problem
$$\left. \begin{matrix}  x^2 \frac{d^2y}{dx^2}+2x\frac{dy}{dx}+\lambda y=0, & x\in [1,2] \\ y(1)=y(2)=0 & \end{matrix} \right \rbrace \quad - (P_2)$$
Which of the following statement is true?

\begin{choices}
\choice There exists a $\lambda_0 \in \mathbb{R}$ such that $(P_\lambda )$ has a nontrivial solution for any $\lambda >\lambda_0$
\choice {$ \lambda \in \mathbb{R} : (P_\lambda ) $  has a nontrivial solution} is a dense subset of $\mathbb{R}$
\choice For any continuous function $f:[1,2] \rightarrow \mathbb{R}$ with $f(x) \neq 0$ for some $x\in [1,2]$, there exists a solution $u$ of $(P_\lambda )$ for some $\lambda \in \mathbb{R}$ such that $\int_{1}^{2}fu\neq 0$
\choice There exists a $\lambda \in \mathbb{R}$ such that $(P_\lambda)$ has two linearly independent solutions. 
\end{choices}

\question 
The system of ODE 

\begin{align*}
\frac{dx}{dt}&=(1+x^2)&&  y,t\in \mathbb{R}&\\ 
\frac{dy}{dt}&=-(1+x^2)&& x,t\in \mathbb{R}&\\ 
(x(0),y(0))&=(a,b) &&&
\end{align*}

has a solution:

\begin{checkboxes}
\choice only if $(a,b)=(0,0)$
\choice For any $(a,b)\in \mathbb{R}\times \mathbb{R}$ 
\choice Such that $x^2(t)+y^2(t)=a^2+b^2$ for all $t\in \mathbb{R}$
\choice Such that $x^2(t)+y^2(t)\rightarrow \infty$ as $t\rightarrow\infty$ if $a>0$ and $b>0$.
\end{checkboxes}

\question 
Let $y:\mathbb{R}\rightarrow\mathbb{R}$ be a solution of the ODE 
$$\frac{d^2y}{dx^2}-y=e^{-x},\quad x\in \mathbb{R}$$
$$y(0)=\frac{dy}{dx}(0) =0$$
then

\begin{checkboxes}
\choice $y$ attains its maximum on $\mathbb{R}$
\choice $y$ is bounded on $\mathbb{R}$
\choice $\lim_{x\to\infty} e^{-x} y(x)=\frac{1}{4}$
\choice $\lim_{x\to\infty} e^{x} y(x)=\frac{1}{4}$
\end{checkboxes}

\question 
Let $P,Q$ be continuous real valued function defined on $[-1,1]$ and $u_i:[-1,1] \rightarrow \mathbb{R},i=1,2$  be solutions of the ODE.
$$\frac{d^2u}{dx^2}+P(x)\frac{du}{dx}+Q(x)u=0,\quad x\in [-1,1]$$
satisfying $u_1\geq 0,u_2\leq 0$ and $u_1(0)=u_2(0)=0$. Let ;$w$ denote the Wronskian of $u_1$ and $u_2$, then 

\begin{checkboxes}
\choice $u_1$ and $u_2$ are linearly independent.
\choice $u_1$ and $u_2$ are linearly independent.
\choice $w(x)=0$ for all $x\in [-1,1]$
\choice $w(x) \neq 0$ for some $x\in [-1,1]$
\end{checkboxes}
